  assets/sounds/

  menu.mp3           — main menu loop (optional)
  ambiance.mp3       — office night loop
  doorclose.mp3
  dooropen.mp3
  camopen.mp3
  camswitch.mp3      — short click when selecting a camera on the map
  camstatic.mp3    — looping CRT/static while camera monitor is open
  powerout.mp3
  scream.mp3         — jumpscare sting
  6am.mp3            — morning win
  Parathalaugh.mp3   — random 1–2× during the night
  phonecall1.mp3     — one call near the start of the shift (single 10‑minute night)
